# How to run?
    
    pip install virtualenv
    virtualenv env
    source env/bin/activate // for linux users
    .\env\Scripts\activate // for windows users
    pip install requirements.txt
    python3 main.py // if you want to train the model again
    python3 test_model.py // before , you should change the index of your webcam
